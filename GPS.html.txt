    <script>
        var map = L.map('map', { zoomControl: false }).setView([46.6, 2.4], 6);
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

        var points = [];
        var control = null;

        function meLocaliser() {
            map.locate({setView: true, maxZoom: 15});
        }

        map.on('locationfound', function(e) {
            if (points.length === 0) {
                points.push(e.latlng);
                L.circle(e.latlng, {radius: 10, color: '#007AFF'}).addTo(map);
                document.getElementById('status').innerText = "üìç Ma position OK. Cliquez l'arriv√©e.";
            }
        });

        map.on('click', function(e) {
            if (points.length < 2) {
                points.push(e.latlng);
                L.marker(e.latlng).addTo(map);
            }
        });

        function calculer() {
            if (points.length < 2) return alert("Positionnez 2 points");
            if (control) map.removeControl(control);

            control = L.Routing.control({
                waypoints: points,
                showAlternatives: true,
                altLineOptions: { styles: [{color: '#000', opacity: 0.1, weight: 12}, {color: 'gray', opacity: 0.6, weight: 6}] },
                lineOptions: { styles: [{color: '#007AFF', weight: 8}] },
                createMarker: function() { return null; }
            }).addTo(map);

            control.on('routesfound', function(e) {
                function update(i) {
                    var r = e.routes[i];
                    document.getElementById('d').innerText = (r.summary.totalDistance/1000).toFixed(1);
                    document.getElementById('t').innerText = Math.round(r.summary.totalTime/60);
                    document.getElementById('status').innerText = "Trajet " + (i+1) + " sur " + e.routes.length;
                }
                update(0);
                control.on('routeSelected', function(ev) {
                    update(e.routes.indexOf(ev.route));
                });
            });
        }
    </script>
